```python
# Filename: series_calculation.py

def calculate_series(terms):
    total = 0.0
    for n in range(terms):
        term = (-1)**n / (2 * n + 1)  # Calculate each term in the series
        total += term  # Accumulate the sum
    return total * 4  # Multiply the total by 4

if __name__ == "__main__":
    num_terms = 10000
    result = calculate_series(num_terms)
    
    # Output the result to a text file
    with open("output.txt", "w") as f:
        f.write(f"The result of calculating the first {num_terms} terms of the series multiplied by 4 is: {result}\n")

```
